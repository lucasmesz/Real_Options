def NPV_EMV(CF, I_w, P_, q, B, I_d):
    V=q*B*P_;
    I_d= I_d*(P_/50) #com preo base
    NPV= V- I_d
    EMV= -I_w + CF*NPV
    
    return [NPV,EMV]

def EMV_portfolio(CF_1, I_w_1, P_1, q_1, B_1, I_d_1, CF_2, I_w_2, P_2, q_2, B_2, I_d_2,rho):
    
    #Mtricas da primeiro prospecto
    V_1=q_1*B_1*P_1;
    I_d_1= I_d_1*(P_1/50) #com de 50 dolar o barril como preo base
    NPV_1= V_1- I_d_1
    EMV_1= -I_w_1 + CF_1*NPV_1
    
    
    #Ajuste do Fator de chance do segundo prospecto
    CF_s_2 = CF_2 + (np.sqrt((1-CF_1)/CF_1))*(np.sqrt(CF_2*(1-CF_2)))*rho
    CF_ins_2 = CF_2 - (np.sqrt(CF_1/(1-CF_1)))*(np.sqrt(CF_2*(1-CF_2)))*rho
    
    #Mtricas para o segundo prospecto, considerando dois cenrios do primeiro poo em Valor Esperado
    V_2=q_2*B_2*P_2;
    I_d_2= I_d_2*(P_2/50) #com preo base
    NPV_2= V_2- I_d_2
    
    EMV_s_2= -I_w_2 + CF_s_2*NPV_2
    EMV_ins_2= -I_w_2 + CF_ins_2*NPV_2
    
    #Anlise de Portfolio, considerando a informao obtida no primeiro poo
        
    portfolio= EMV_1 + CF_1*max(EMV_s_2,0)+ (1-CF_1)*max(EMV_ins_2,0)
    
    return [portfolio]

def defer_seq(case,r,delta,sigma,delta_t,T,Po,rho):

    #def defer_seq_learning_option(r,Po,delta,sigma,delta_t,T):
    # Mtodo binomial Cox-Ross-Rubinstein - Baseado nos exerccios do Marco Antonio e Captilo 9 do primeiro livro

    #up, down e medida neutra ao risco
    u=np.exp(sigma*np.sqrt(delta_t))
    d=1/u
    q= ((np.exp((r-delta)*delta_t) -d))/(u-d)

    P_m=[]
    P_m_1=[]
    P_m.append(Po)
    for passo in np.arange(T*(1/delta_t)):
        for i in range(len(P_m)):
       
            Pu=P_m[i]*u
            Pd=P_m[i]*d
            P_m_1.append(Pu)
            P_m_1.append(Pd)
        P_m=[]
        P_m=P_m_1 #Registro Final est em P_m . 
        P_m_1=[]
        #print(P_m)
    
    safe=[*case.Safe]
    risky=[*case.Risky]
    del safe[2]
    del risky[2]



    #deciso e Melhor escolha no vencimento
    port=[]
    decision=[]
    for i in range(len(P_m)):
        #lista com a melhor escolha entre SR,RS e abandono=0
        #lista com tamanho de P_m
        W=max(*EMV_portfolio_P(*safe, *risky,P_m[i],rho),*EMV_portfolio_P(*risky,*safe,P_m[i],rho),0)
        port.append(W)
    
        if W <= 0:
            decision.append('AB')
        elif W== EMV_portfolio_P(*safe, *risky,P_m[i],rho):
            decision.append('SR')
        else:
            decision.append('RS')
    #    print(W)
    #print(decision)


    for passo in np.arange(T*(1/delta_t)):
        P_m_1=[]
    
        for i in range(int(len(P_m)/2)):
            P_m_1.append(P_m[i]/u)
     
        #tanto port_1 e decision_1 so informaes que so apagadas a cada loop, sobrando somente a ltima, ou seja,
        #o valor da opo/portfolio hoje e da deciso
        port_1=[]
        decision_1=[]
        for i in range(len(P_m_1)):
                 #lista com tamanho de P_m_1

                W=max (*EMV_portfolio_P(*safe, *risky,P_m_1[i],rho),*EMV_portfolio_P(*risky,*safe,P_m_1[i],rho),
